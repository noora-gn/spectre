24 // Perform an Arnoldi orthogonalization to find a new `operand` that is
25 // orthogonal to all vectors in `basis_history`. Appends a new column to the
26 // `orthogonalization_history` that holds the inner product of the intermediate
27 // `operand` with each vector in the `basis_history` and itself.
28 template <typename VarsType>
29 void arnoldi_orthogonalize(
30  const gsl::not_null<VarsType*> operand,
31  const gsl::not_null<DenseMatrix<double>*> orthogonalization_history,
32  const std::vector<VarsType>& basis_history,
33  const size_t iteration) noexcept {
34  // Resize matrix and make sure the new entries that are not being filled below
35  // are zero.
36  orthogonalization_history->resize(iteration + 2, iteration + 1);
37  for (size_t j = 0; j < iteration; ++j) {
38  (*orthogonalization_history)(iteration + 1, j) = 0.;
39  }
40  // Arnoldi orthogonalization
41  for (size_t j = 0; j < iteration + 1; ++j) {
42  const double orthogonalization = inner_product(basis_history[j], *operand);
43  (*orthogonalization_history)(j, iteration) = orthogonalization;
44  *operand -= orthogonalization * basis_history[j];
45  }
46  (*orthogonalization_history)(iteration + 1, iteration) =
47  sqrt(inner_product(*operand, *operand));
48  // Avoid an FPE if the new operand norm is exactly zero. In that case the
49  // problem is solved and the algorithm will terminate (see Proposition 9.3 in
50  // \cite Saad2003). Since there will be no next iteration we don't need to
51  // normalize the operand.
52  if (UNLIKELY((*orthogonalization_history)(iteration + 1, iteration) == 0.)) {
53  return;
54  }
55  *operand /= (*orthogonalization_history)(iteration + 1, iteration);
56 }

89 /*!
90  * \brief A serial GMRES iterative solver for nonsymmetric linear systems of
91  * equations.
92  *
93  * This is an iterative algorithm to solve general linear equations \f$Ax=b\f$
94  * where \f$A\f$ is a linear operator. See \cite Saad2003, chapter 6.5 for a
95  * description of the GMRES algorithm and Algorithm 9.6 for this implementation.
96  * It is matrix-free, which means the operator \f$A\f$ needs not be provided
97  * explicity as a matrix but only the operator action \f$A(x)\f$ must be
98  * provided for an argument \f$x\f$.
99  *
100  * The GMRES algorithm does not require the operator \f$A\f$ to be symmetric or
101  * positive-definite. Note that other algorithms such as conjugate gradients may
102  * be more efficient for symmetric positive-definite operators.
103  *
104  * \par Convergence:
105  * Given a set of \f$N_A\f$ equations (e.g. through an \f$N_A\times N_A\f$
106  * matrix) the GMRES algorithm will converge to numerical precision in at most
107  * \f$N_A\f$ iterations. However, depending on the properties of the linear
108  * operator, an approximate solution can ideally be obtained in only a few
109  * iterations. See \cite Saad2003, section 6.11.4 for details on the convergence
110  * of the GMRES algorithm.
111  *
112  * \par Restarting:
113  * This implementation of the GMRES algorithm supports restarting, as detailed
114  * in \cite Saad2003, section 6.5.5. Since the GMRES algorithm iteratively
115  * builds up an orthogonal basis of the solution space the cost of each
116  * iteration increases linearly with the number of iterations. Therefore it is
117  * sometimes helpful to restart the algorithm every \f$N_\mathrm{restart}\f$
118  * iterations, discarding the set of basis vectors and starting again from the
119  * current solution estimate. This strategy can improve the performance of the
120  * solver, but note that the solver can stagnate for non-positive-definite
121  * operators and is not guaranteed to converge within \f$N_A\f$ iterations
122  * anymore. Set the `restart` argument of the constructor to
123  * \f$N_\mathrm{restart}\f$ to activate restarting, or set it to zero to
124  * deactivate restarting (default behaviour).
125  *
126  * \par Preconditioning:
127  * This implementation of the GMRES algorithm also supports preconditioning.
128  * You can provide a linear operator \f$P\f$ that approximates the inverse of
129  * the operator \f$A\f$ to accelerate the convergence of the linear solve.
130  * The algorithm is right-preconditioned, which allows the preconditioner to
131  * change in every iteration ("flexible" variant). See \cite Saad2003, sections
132  * 9.3.2 and 9.4.1 for details. This implementation follows Algorithm 9.6 in
133  * \cite Saad2003.
134  *
135  * \par Improvements:
136  * Further improvements can potentially be implemented for this algorithm, see
137  * e.g. \cite Ayachour2003.
138  *
139  * \example
140  * \snippet NumericalAlgorithms/LinearSolver/Test_Gmres.cpp gmres_example
141  */

214  /*!
215  * \brief Iteratively solve the problem \f$Ax=b\f$ for \f$x\f$ where \f$A\f$
216  * is the `linear_operator` and \f$b\f$ is the `source`, starting \f$x\f$ at
217  * `initial_guess`.
218  *
219  * Optionally provide a `preconditioner` (see class documentation).
220  *
221  * \return An instance of `Convergence::HasConverged` that provides
222  * information on the convergence status of the completed solve, and the
223  * approximate solution \f$x\f$.
224  */
225  template <typename LinearOperator, typename SourceType,
226  typename Preconditioner = IdentityPreconditioner<VarsType>,
227  typename IterationCallback = NoIterationCallback>
228  std::pair<Convergence::HasConverged, VarsType> operator()(
229  LinearOperator&& linear_operator, const SourceType& source,
230  const VarsType& initial_guess,
231  Preconditioner&& preconditioner = IdentityPreconditioner<VarsType>{},
232  IterationCallback&& iteration_callback = NoIterationCallback{}) const
233  noexcept;
234 
235  private:
236  Convergence::Criteria convergence_criteria_{};
237  ::Verbosity verbosity_{::Verbosity::Verbose};
238  size_t restart_{};
239 
240  // Memory buffers to avoid re-allocating memory for successive solves:
241  // The `orthogonalization_history_` is built iteratively from inner products
242  // between existing and potential basis vectors and then Givens-rotated to
243  // become upper-triangular.
244  mutable DenseMatrix<double> orthogonalization_history_{};
245  // The `residual_history_` holds the remaining residual in its last entry, and
246  // the other entries `g` "source" the minimum residual vector `y` in
247  // `R * y = g` where `R` is the upper-triangular `orthogonalization_history_`.
248  mutable DenseVector<double> residual_history_{};
249  // These represent the accumulated Givens rotations up to the current
250  // iteration.
251  mutable DenseVector<double> givens_sine_history_{};
252  mutable DenseVector<double> givens_cosine_history_{};
253  // These represent the orthogonal Krylov-subspace basis that is constructed
254  // iteratively by Arnoldi-orthogonalizing a new vector in each iteration and
255  // appending it to the `basis_history_`.
256  mutable std::vector<VarsType> basis_history_{};
257  // When a preconditioner is used it is applied to each new basis vector. The
258  // preconditioned basis is used to construct the solution when the algorithm
259  // has converged.
260  mutable std::vector<VarsType> preconditioned_basis_history_{};
261 };
262 
263 template <typename VarsType>
264 template <typename LinearOperator, typename SourceType, typename Preconditioner,
265  typename IterationCallback>
266 std::pair<Convergence::HasConverged, VarsType> Gmres<VarsType>::operator()(
267  LinearOperator&& linear_operator, const SourceType& source,
268  const VarsType& initial_guess, Preconditioner&& preconditioner,
269  IterationCallback&& iteration_callback) const noexcept {
270  constexpr bool use_preconditioner =
271  not std::is_same_v<Preconditioner, IdentityPreconditioner<VarsType>>;
272  constexpr bool use_iteration_callback =
273  not std::is_same_v<IterationCallback, NoIterationCallback>;
274 
275  auto result = initial_guess;
276  Convergence::HasConverged has_converged{};
277  size_t iteration = 0;
278 
279  while (not has_converged) {
280  auto& initial_operand = basis_history_[0] =
281  source - linear_operator(result);
282  const double initial_residual_magnitude =
283  sqrt(inner_product(initial_operand, initial_operand));
284  has_converged = Convergence::HasConverged{convergence_criteria_, iteration,
285  initial_residual_magnitude,
286  initial_residual_magnitude};
287  if (use_iteration_callback) {
288  iteration_callback(has_converged);
289  }
290  if (UNLIKELY(has_converged)) {
291  break;
292  }
293  initial_operand /= initial_residual_magnitude;
294  residual_history_.resize(1);
295  residual_history_[0] = initial_residual_magnitude;
296  for (size_t k = 0; k < restart_; ++k) {
297  auto& operand = basis_history_[k + 1];
298  if (use_preconditioner) {
299  preconditioned_basis_history_[k] = preconditioner(basis_history_[k]);
300  }
301  operand =
302  linear_operator(use_preconditioner ? preconditioned_basis_history_[k]
303  : basis_history_[k]);
304  // Find a new orthogonal basis vector of the Krylov subspace
305  gmres::detail::arnoldi_orthogonalize(
306  make_not_null(&operand), make_not_null(&orthogonalization_history_),
307  basis_history_, k);
308  // Least-squares solve for the minimal residual
309  gmres::detail::solve_minimal_residual(
310  make_not_null(&orthogonalization_history_),
311  make_not_null(&residual_history_),
312  make_not_null(&givens_sine_history_),
313  make_not_null(&givens_cosine_history_), k);
314  ++iteration;
315  has_converged = Convergence::HasConverged{
316  convergence_criteria_, iteration, abs(residual_history_[k + 1]),
317  initial_residual_magnitude};
318  if (use_iteration_callback) {
319  iteration_callback(has_converged);
320  }
321  if (UNLIKELY(has_converged)) {
322  break;
323  }
324  }
325  // Find the vector w.r.t. the constructed orthogonal basis of the Krylov
326  // subspace that minimizes the residual
327  const auto minres = gmres::detail::minimal_residual_vector(
328  orthogonalization_history_, residual_history_);
329  // Construct the solution from the orthogonal basis and the minimal residual
330  // vector
331  for (size_t i = 0; i < minres.size(); ++i) {
332  result +=
333  minres[i] * gsl::at(use_preconditioner ? preconditioned_basis_history_
334  : basis_history_,
335  i);
336  }
337  }
338  // NOLINTNEXTLINE(hicpp-move-const-arg,performance-move-const-arg)
339  return {std::move(has_converged), std::move(result)};
340 }
