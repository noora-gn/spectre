60 template <size_t I, size_t Index1, size_t Index2, typename... LhsIndices,
61  typename T, typename S>
62 static SPECTRE_ALWAYS_INLINE decltype(auto) compute_contraction(S tensor_index,
63  const T& t1) {
64  if constexpr (I == 0) {
65  tensor_index[Index1] = 0;
66  tensor_index[Index2] = 0;
67  return t1.template get<LhsIndices...>(tensor_index);
68  } else {
69  tensor_index[Index1] = I;
70  tensor_index[Index2] = I;
71  return t1.template get<LhsIndices...>(tensor_index) +
72  compute_contraction<I - 1, Index1, Index2, LhsIndices...>(
73  tensor_index, t1);
74  }
75 }

98  // First and second \ref SpacetimeIndex "TensorIndexType"s to contract.
99  // "first" and "second" here refer to the position of the indices to contract
100  // in the list of indices, with "first" denoting leftmost
101  //
102  // e.g. `R(ti_A, ti_b, ti_a)` :
103  // - `first_contracted_index` refers to the
104  // \ref SpacetimeIndex "TensorIndexType" refered to by `ti_A`
105  // - `second_contracted_index` refers to the
106  // \ref SpacetimeIndex "TensorIndexType" refered to by `ti_a`
107  using first_contracted_index = tmpl::at_c<IndexList, FirstContractedIndexPos>;
108  using second_contracted_index =
109  tmpl::at_c<IndexList, SecondContractedIndexPos>;

111  tmpl::size<IndexList>::value > 1,
112  "Cannot contract indices on a Tensor with rank less than 2");
113  static_assert(detail::indices_contractible<first_contracted_index,
114  second_contracted_index>::value,
115  "Cannot contract the requested indices.");

132  template <size_t I, size_t Rank>
133  SPECTRE_ALWAYS_INLINE void fill_contracting_tensor_index(
134  std::array<size_t, Rank>& tensor_index_in,
135  const std::array<size_t, num_tensor_indices>& tensor_index) const {
136  if constexpr (I < FirstContractedIndexPos) {
137  tensor_index_in[I] = tensor_index[I];
138  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
139  } else if constexpr (I == FirstContractedIndexPos) {
140  // 10000 is for the slot that will be set later. Easy to debug.
141  tensor_index_in[I] = 10000;
142  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
143  } else if constexpr (I > FirstContractedIndexPos and
144  I <= SecondContractedIndexPos and I < Rank - 1) {
145  // tensor_index is Rank - 2 since it shouldn't be called for Rank 2 case
146  // 20000 is for the slot that will be set later. Easy to debug.
147  tensor_index_in[I] =
148  I == SecondContractedIndexPos ? 20000 : tensor_index[I - 1];
149  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
150  } else if constexpr (I > SecondContractedIndexPos and I < Rank - 1) {
151  // Left as Rank - 2 since it should never be called for the Rank 2 case
152  tensor_index_in[I] = tensor_index[I - 2];
153  fill_contracting_tensor_index<I + 1>(tensor_index_in, tensor_index);
154  } else if constexpr (I == SecondContractedIndexPos) {
155  tensor_index_in[I] = 20000;
156  } else {
157  tensor_index_in[I] = tensor_index[I - 2];
158  }
159  }

162  SPECTRE_ALWAYS_INLINE decltype(auto) get(
163  const std::array<U, num_tensor_indices>& new_tensor_index) const {
164  // new_tensor_index is the one with _fewer_ components, ie post-contraction

190 /*!
191  * \ingroup TensorExpressionsGroup
192  * \brief Returns the positions of the first indices to contract in an
193  * expression
194  *
195  * \details Given a list of values that represent an expression's generic index
196  * encodings, this function looks to see if it can find a pair of values that
197  * encode one generic index and the generic index with opposite valence, such as
198  * `ti_A_t` and `ti_a_t`. This denotes a pair of indices that will need to be
199  * contracted. If there exists more than one such pair of indices in the
200  * expression, the first pair of values found will be returned.
201  *
202  * For example, if we have tensor \f${R^{ab}}_{ab}\f$ represented by the tensor
203  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then this will return the positions
204  * of the pair of values encoding `ti_A_t` and `ti_a_t`, which would be (0, 2)
205  *
206  * @param tensorindex_values the TensorIndex values of a tensor expression
207  * @return the positions of the first pair of TensorIndex values to contract
208  */
209 template <size_t NumIndices>
210 SPECTRE_ALWAYS_INLINE static constexpr std::pair<size_t, size_t>
211 get_first_index_positions_to_contract(
212  const std::array<size_t, NumIndices>& tensorindex_values) noexcept {
213  for (size_t i = 0; i < tensorindex_values.size(); ++i) {
214  const size_t current_value = gsl::at(tensorindex_values, i);
215  const size_t opposite_value_to_find =
216  get_tensorindex_value_with_opposite_valence(current_value);
217  for (size_t j = i + 1; j < tensorindex_values.size(); ++j) {
218  if (opposite_value_to_find == gsl::at(tensorindex_values, j)) {
219  // We found both the lower and upper version of a generic index in the
220  // list of generic indices, so we return this pair's positions
221  return std::pair{i, j};
222  }
223  }
224  }
225  // We couldn't find a single pair of indices that needs to be contracted
226  return std::pair{std::numeric_limits<size_t>::max(),
227  std::numeric_limits<size_t>::max()};
228 }

230 /*!
231  * \ingroup TensorExpressionsGroup
232  * \brief Creates a contraction expression from a tensor expression if there are
233  * any indices to contract
234  *
235  * \details If there are no indices to contract, the input TensorExpression is
236  * simply returned. Otherwise, a contraction expression is created for
237  * contracting one pair of upper and lower indices. If there is more than one
238  * pair of indices to contract, subsequent contraction expressions are
239  * recursively created, nesting one contraction expression inside another.
240  *
241  * For example, if we have tensor \f${R^{ab}}_{ab}\f$ represented by the tensor
242  * expression, `R(ti_A, ti_B, ti_a, ti_b)`, then one contraction expression is
243  * created to represent contracting \f${R^{ab}}_ab\f$ to \f${R^b}_b\f$, and a
244  * second to represent contracting \f${R^b}_b\f$ to the scalar, \f${R}\f$.
245  *
246  * @param t the TensorExpression to potentially contract
247  * @return the input tensor expression or a contraction expression of the input
248  */
249 template <typename T, typename X, typename Symm, typename IndexList,
250  typename... TensorIndices>
251 SPECTRE_ALWAYS_INLINE static constexpr auto contract(
252  const TensorExpression<T, X, Symm, IndexList, tmpl::list<TensorIndices...>>&
253  t) noexcept {
254  constexpr std::array<size_t, sizeof...(TensorIndices)> tensorindex_values = {
255  {TensorIndices::value...}};
256  constexpr std::pair first_index_positions_to_contract =
257  get_first_index_positions_to_contract(tensorindex_values);
258  constexpr std::pair no_indices_to_contract_sentinel{
259  std::numeric_limits<size_t>::max(), std::numeric_limits<size_t>::max()};
260 
261  if constexpr (first_index_positions_to_contract ==
262  no_indices_to_contract_sentinel) {
263  // There aren't any indices to contract, so we just return the input
264  return ~t;
265  } else {
266  // We have a pair of indices to be contract
267  return contract(
268  TensorContract<first_index_positions_to_contract.first,
269  first_index_positions_to_contract.second, T, X, Symm,
270  IndexList, tmpl::list<TensorIndices...>>{t});
271  }
272 }
