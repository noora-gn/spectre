86 /*!
87  * Compute the Newtonian sound speed squared
88  * \f$c_s^2 = \chi + p\kappa / \rho^2\f$, where
89  * \f$p\f$ is the fluid pressure, \f$\rho\f$ is the mass density,
90  * \f$\chi = (\partial p/\partial\rho)_\epsilon\f$ and
91  * \f$\kappa = (\partial p/ \partial \epsilon)_\rho\f$, where
92  * \f$\epsilon\f$ is the specific internal energy.
93  */
94 template <typename DataType, size_t ThermodynamicDim>
95 void sound_speed_squared(
96  gsl::not_null<Scalar<DataType>*> result,
97  const Scalar<DataType>& mass_density,
98  const Scalar<DataType>& specific_internal_energy,
99  const EquationsOfState::EquationOfState<false, ThermodynamicDim>&
100  equation_of_state) noexcept;
101
