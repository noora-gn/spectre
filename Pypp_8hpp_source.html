25 template <typename R, typename = std::nullptr_t>
26 struct CallImpl {
27  template <typename... Args>
28  static R call(const std::string& module_name,
29  const std::string& function_name, const Args&... t) {
30  PyObject* module = PyImport_ImportModule(module_name.c_str());
31  if (module == nullptr) {
32  PyErr_Print();
33  throw std::runtime_error{std::string("Could not find python module.\n") +
34  module_name};
35  }
36  PyObject* func = PyObject_GetAttrString(module, function_name.c_str());
37  if (func == nullptr or not PyCallable_Check(func)) {
38  if (PyErr_Occurred()) {
39  PyErr_Print();
40  }
41  throw std::runtime_error{"Could not find python function in module.\n"};
42  }
43  PyObject* args = pypp::make_py_tuple(t...);
44  PyObject* value = PyObject_CallObject(func, args);
45  Py_DECREF(args); // NOLINT
46  if (value == nullptr) {
47  Py_DECREF(func); // NOLINT
48  Py_DECREF(module); // NOLINT
49  PyErr_Print();
50  throw std::runtime_error{"Function returned null"};
51  }
52  auto ret = from_py_object<R>(value);
53  Py_DECREF(value); // NOLINT
54  Py_DECREF(func); // NOLINT
55  Py_DECREF(module); // NOLINT
56  return ret;
57  }
58 };

94 template <typename T>
95 struct SliceContainerImpl {
96  static auto apply(const T& container_dv, const size_t slice_idx) noexcept {
97  convert_to_container_of_doubles_t<std::decay_t<decltype(container_dv)>>
98  container_double{};
99  ASSERT(slice_idx < container_dv.begin()->size(),
100  "Trying to slice DataVector of size " << container_dv.begin()->size()
101  << "with slice_idx "
102  << slice_idx);
103  for (decltype(auto) double_and_datavector_components :
104  boost::combine(container_double, container_dv)) {
105  boost::get<0>(double_and_datavector_components) =
106  boost::get<1>(double_and_datavector_components)[slice_idx];
107  }
108  return container_double;
109  }
110 };

285 /// Calls a Python function from a module/file with given parameters
286 ///
287 /// \param module_name name of module the function is in
288 /// \param function_name name of Python function in module
289 /// \param t the arguments to be passed to the Python function
290 /// \return the object returned by the Python function converted to a C++ type
291 ///
292 /// Custom classes can be converted between Python and C++ by overloading the
293 /// `pypp::ToPythonObject<T>` and `pypp::FromPythonObject<T>` structs for your
294 /// own types. This tells C++ how to deconstruct the Python object into
295 /// fundamental types and reconstruct the C++ object and vice-versa.
296 ///
297 /// \note In order to setup the python interpreter and add the local directories
298 /// to the path, a SetupLocalPythonEnvironment object needs to be constructed
299 /// in the local scope.
300 ///
301 /// \example
302 /// The following example calls the function `test_numeric` from the module
303 /// `pypp_py_tests` which multiplies two integers.
304 /// \snippet Test_Pypp.cpp pypp_int_test
305 /// Alternatively, this examples calls `test_vector` from `pypp_py_tests` which
306 /// converts two vectors to python lists and multiplies them pointwise.
307 /// \snippet Test_Pypp.cpp pypp_vector_test
308 ///
309 /// Pypp can also be used to take a function that performs manipulations of
310 /// NumPy arrays and apply it to either a Tensor of doubles or a Tensor of
311 /// DataVectors. This is useful for testing functions which act on Tensors
312 /// pointwise. For example, let's say we wanted to call the NumPy function which
313 /// performs \f$ v_i = A B^a C_{ia} + D^{ab} E_{iab} \f$, which is implemented
314 /// in python as
315 ///
316 /// \code{.py} def test_einsum(scalar, t_A, t_ia, t_AA, t_iaa):
317 /// return scalar * np.einsum("a,ia->i", t_A, t_ia) +
318 /// np.einsum("ab, iab->i", t_AA, t_iaa)
319 /// \endcode
320 ///
321 /// where \f$ v_i \f$ is the return tensor and
322 /// \f$ A, B^a, C_{ia},D^{ab}, E_{iab} \f$ are the input tensors respectively.
323 /// We call this function through C++ as:
324 /// \snippet Test_Pypp.cpp einsum_example
325 /// for type `T` either a `double` or `DataVector`.
326 ///
327 /// Pypp will also support testing of functions which return and operate on
328 /// `std::array`s of `DataVectors`s. To return a `std::array` of DataVectors,
329 /// the python function should return a python list of doubles.
330 ///
331 /// \note In order to return a
332 /// `Tensor<DataVector...>` from `pypp::call`, at least one
333 /// `Tensor<DataVector...>` must be taken as an argument, as the size of the
334 /// returned tensor needs to be deduced.
335 template <typename R, typename... Args>
336 R call(const std::string& module_name, const std::string& function_name,
337  const Args&... t) {
338  return detail::CallImpl<R>::call(module_name, function_name, t...);
339 }
